La ce ne ajută să știm timpul de execuție?

  • Putem crea o statistică a timpilor de execuție a unor algoritmi ce execută același task (de exemplu sortarea unui vector).
  • Putem verifica diferența dintre timpul necesar pentru a rezolva problema și timpul curent al programului.
  • Există și alte moduri de a măsura timpul - Metodă alternativă 
#include <iostream>
#include <Windows.h>
using namespace std;

static inline int64_t GetTicks() {
    LARGE_INTEGER ticks;
    if (!QueryPerformanceCounter(&ticks)) {
        // Handle error (you can throw an exception or handle it differently)
    }
    return ticks.QuadPart;
}

int main() {
    int64_t start = GetTicks();

    /// Aici pui algoritmul / apelul subprogramului pentru care contorizezi timpul

    int64_t end = GetTicks();

    double elapsedTimeSeconds = (double)(end - start);

    cout << "Elapsed time: " << fixed << setprecision(3) << elapsedTimeSeconds / 1e7 << " seconds";

    return 0;
}
